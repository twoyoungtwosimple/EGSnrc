31 # A general dose calculation tool. Since it is outside the scope of the
32 # EGS_Application class, a few utility methods are needed in EGS_Application
33 # to retrieve information regarding media, geometry and source.
34 #
35 # Dose is output for each dose region without any spacial information. For a
36 # more detailed dose scoring grid it is better to create a user code which
37 # could produce 3D dose distributions and graph data.
38 #
39 # Can be used in any C++ user code by entering the proper input block in
40 # the ausgab object definition block.
41 #
42 # Prints out dose in each dose scoring region.
43 #
44 # Dose scoring regions can be defined by input using one the following
45 # syntax for individual regions:
46 #
47 # :start ausgab object definition:
48 # :start ausgab object:
49 # library = egs_dose_scoring
50 # name = some_name
51 # volume = V0 V1 ... Vn
52 # dose regions = IR0 IR1 ... IRn
53 # :stop ausgab object:
54 # :stop ausgab object definition:
55 #
56 # or by groups of consecutive regions:
57 #
58 # :start ausgab object definition:
59 # :start ausgab object:
60 # library = egs_dose_scoring
61 # name = some_name
62 # volume = V0 V1 ... Vn
63 # dose start region = IRI_0 ... IRI_N
64 # dose stop region = IRE_0 ... IRE_N
65 # :stop ausgab object:
66 # :stop ausgab object definition:
67 #
68 # where one can have same number of volume entries as group of regions (n=N)
69 # or same number of volume entries as number of individual regions. If there
70 # are more dose scoring regions than volume entries, a default volume is use for
71 # those regions without volume information. This default can be either the first
72 # volume entry or 1 g/cm3.
73 #
74 # TODO:
75 #
76 # - Classify dose as contributed by primary or scattered particles
77 # - Custom classification using latch
78 # - Dose to medium calculation
79 #
80 ###############################################################################
81 */

252 void EGS_DoseScoring::reportResults() {

254  egsInformation("Dose Scoring Object(%s)\n",name.c_str());
255  egsInformation("======================================================\n");
256  EGS_Float normD = 1., normE=1.;
257  int count = 0;
258  EGS_Float F = app->getFluence();
259  egsInformation("=> last case = %lld fluence = %g\n", m_lastCase, F);
260  /* Normalize to actual source fluence */
261  normE = m_lastCase/F*norm_u;
262  normD = 1.602e-10*normE;
263  int irmax_digits = getDigits(max_dreg);
264  string line;
265  double r,dr;
266  if (dose) {
267  if (normE==1) {
268  egsInformation("\n\n==> Summary of region dosimetry (per particle)\n");
269  egsInformation(
270  "%*s %*s %*s rho/[g/cm3] V/cm3 Edep/[MeV] D/[Gy] %n\n",
271  irmax_digits,"ir",max_medl/2,"medium",max_medl/2," ",&count);
272  }
273  else {
274  egsInformation("\n==> Summary of region dosimetry (per fluence)\n");
275  egsInformation(
276  "%*s %*s %*s rho/[g/cm3] V/cm3 Edep/[MeV*cm2] D/[Gy*cm2] %n\n",
277  irmax_digits,"ir",max_medl/2,"medium",max_medl/2," ",&count);
278  }
279  line.append(count,'-');
280  egsInformation("%s\n",line.c_str());
281 
282  /* Compute deposited energy and dose */
283  for (int ireg = 0; ireg < nreg; ireg++) {
284  if (d_reg_index[ireg]>=0) {
285  if (!(app->isRealRegion(ireg))) {
286  continue;
287  }
288  int imed = app->getMedium(ireg);
289  EGS_Float rho = app->getMediumRho(imed);
290  EGS_Float mass = vol[ireg]*rho;
291  dose->currentResult(d_reg_index[ireg],r,dr);
292  if (r > 0) {
293  dr = dr/r;
294  }
295  else {
296  dr=1;
297  }
298  egsInformation("%*d %-*s %7.3f %8.4f %10.4e +/- %-7.3f%% %10.4e +/- %-7.3f%%\n",

300  }
301  }
302  egsInformation("%s\n",line.c_str());
303  }
304  if (doseM) {
305  vector<EGS_Float> massM(nmedia,0);
306  int imed = 0;
307  for (int ir=0; ir<nreg; ir++) {
308  if (app->isRealRegion(ir)) {
309  imed = app->getMedium(ir);
310  EGS_Float volume = vol.size() > 1 ? vol[ir]:vol[0];
311  massM[imed] += app->getMediumRho(imed)*volume;
312  }
313  }
314  if (normE==1) {
315  egsInformation("\n\n==> Summary of media dosimetry (per particle)\n");
316  egsInformation(
317  "%*s %*s Edep/[MeV] D/[Gy] %n\n",
318  max_medl/2,"medium",max_medl/2," ",&count);
319  }
320  else {
321  egsInformation("\n\n==> Summary of media dosimetry (per fluence)\n");
322  egsInformation(
323  "%*s %*s Edep/[MeV*cm2] D/[Gy*cm2] %n\n",
324  max_medl/2,"medium",max_medl/2," ",&count);
325  }
326  line="";
327  line.append(count,'-');
328  egsInformation("%s\n",line.c_str());

331  doseM->currentResult(im,r,dr);
332  if (r > 0) {
333  dr = dr/r;
334  egsInformation(

413 //*********************************************************************
414 // Process input for this ausgab object
415 //
416 // volume: Set to 1 cm3 by default
417 // One entry => same size dose scoring regions
418 // Several entries => volume for each dose region
419 // SAME as dose region number
420 // or else default value used
421 //
422 // dose scoring regions: Defaults to all regions in geometry
423 // dose regions => Individual regions
424 // dose start/stop region => Groups of consecutive
425 // dose regions
426 //
427 // If there is a mismatch between number of regions and volumes
428 // default volume values will be used. Default volume will be either
429 // the first volume entry or 1 g/cm3 if no volume entry found.
430 //
431 // TODO:
432 // - Classify in primary, scattered and total dose
433 // - Specify for wich media to score or not the dose
434 //
435 //**********************************************************************

438  EGS_DOSE_SCORING_EXPORT EGS_AusgabObject *createAusgabObject(EGS_Input *input,
439  EGS_ObjectFactory *f) {
440  const static char *func = "createAusgabObject(dose_scoring)";
441  if (!input) {
442  egsWarning("%s: null input?\n",func);
443  return 0;
444  }
445  vector <EGS_Float> v_in;// voxel volume[s] in g/cm3
446  /* get voxel volume */
447  input->getInput("volume",v_in);
448  /* get dose scoring mode: region dose, medium dose or both */
449  vector<string> allowed_mode;
450  allowed_mode.push_back("no");
451  allowed_mode.push_back("yes");
452  int d_in_medium = input->getInput("medium dose",allowed_mode,0);
453  int d_in_region = input->getInput("region dose",allowed_mode,1);
454 
455  /* get dose regions */
456  string d_regionsString;
457  vector <int> d_regions;
458  bool using_all_regions=true;
459  vector <int> d_start, d_stop;
460  if (!input->getInput("dose regions",d_regionsString)&& d_regionsString.length()>0) {
461  using_all_regions = false; // individual regions
462  }
463  else {
464  int err1 = input->getInput("dose start region",d_start);
465  int err2 = input->getInput("dose stop region",d_stop);
466  if (!err1 && !err2) {
467  if (d_start.size()==d_stop.size()) { // group of dose regions
468  for (int i=0; i<d_start.size(); i++) {
469  int ir = d_start[i], fr = d_stop[i];
470  for (int ireg=ir; ireg<=fr; ireg++) {
471  d_regions.push_back(ireg);
472  }
473  }
474  using_all_regions = false;
475  }
476  else egsWarning(
477  "%s: Mismatch in start and stop dose region groups !!!\n"
478  " Calculating dose in ALL regions.\n",func);
479  }
480  }
481  EGS_Float norma = 1.0;
482  int err04 = input->getInput("normalization",norma);
483 
484  //================================================
485  // Check if one volume for each group requested.
486  // If not just pass the volumes read and if there
487  // is a mismatch, then the first volume element
488  // or 1g/cm3 will be used.
489  //=================================================
490  vector <EGS_Float> volin;
491  // groups of regions with same volume
492  if (! using_all_regions && v_in.size()== d_start.size()) {
493  for (int i=0; i<d_start.size(); i++) {
494  int ir = d_start[i], fr = d_stop[i];
495  for (int ireg=ir; ireg<=fr; ireg++) {
496  volin.push_back(v_in[i]);
497  }
498  }
499  }
500  else { // all other possibilities handled here
501  volin = v_in;
502  }
503  //=================================================
504 
505  /* Setup dose scoring object with input parameters */
506  EGS_DoseScoring *result = new EGS_DoseScoring("",f);
507  if (volin.size()==1) {
508  result->setVol(volin[0]); // one size for all regions
509  }
510  else if (volin.size()) {
511  result->setVol(volin); // regions with their volumes
512  }
513  else {
514  result->setVol(1.0); // default value if no entry
515  }
516  if (!using_all_regions) {
517  if (d_regions.size() > 0) {
518  result->setDoseRegions(d_regions);
519  }
520  else {
521  result->setDoseRegions(d_regionsString);
522  }
523  }
524  if (d_in_medium) {
525  result->setMediumScoring(true);
526  }
527  if (d_in_region) {
528  result->setRegionScoring(true);
529  }
530  result->setName(input);
531  if (!err04) {
532  result->setUserNorm(norma);
533  }
534  return result;
535  }
